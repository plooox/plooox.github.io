print(x_label)
print("\n[Y_Label]")
print(y_label)

테스트 데이터 분리

x와 y가 이미 분리된 데이터에 대한 테스트 데이터 분리 과정

sklearn

train_test_split을 활용해 테스트 데이터 분리

1

x_train, x_test, y_train, y_test = train_test_split(x, y, test_size= 0.2, random_state=42)
  • x: 독립 변수 데이터(배열, 데이터 프레임)
  • y: 종속 변수 데이터, 레이블 데이터
  • test_size: 테스트용 데이터 개수, 1보다 작을 시 비율 의미
  • train_size: 학습용 데이터의 개수, 1바도 작을 시 비율 의미
  • random_state: 난수 시드

import numpy as np
from sklearn.model_selection import train_test_split
x,y = np.arange(10).reshape((5,2)), range(5) #분리된 x,y 데이터 생성
y = list(y)
print(x)
print(y)
print("\n\n")
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size = 0.33, random_state = 1234)
# 전체 데이터 중 1/3을 테스트 데이터로 지정
# random_state의 seed_value = 42

print("[X_train]")
print(x_train)
print("[X_test]")
print(x_test)

print("\n[Y_train]")
print(y_train)
print("[Y_test]")
print(y_test)
